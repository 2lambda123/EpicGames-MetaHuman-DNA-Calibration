20// ArchiveSize is a type that stores an arbitrary size value (both in-memory and writes that value into the stream as well).
21// It has two accompanying types: Proxy and Anchor, which are both virtual, utility types, and they work together to achieve
22// the function of ArchiveSize.
23// When ArchiveSize is encountered during the serialization process, at first an empty size (zero) value will be written
24// into the stream and the position of that size value will be captured in an in-memory variable.
25// Later, when its associated Anchor is encountered, it will capture the current stream position as the base (start) offset,
26// from which the size is going to be computed (size is relative to some base offset).
27// Lastly, when its ArchiveSize<T>::Proxy is encountered, the current stream position will be captured, and the earlier captured
28// anchor base offset will be subtracted from it (resulting in the size value - the size between Anchor offset and Proxy offset).
29// With the size value computed, it will update the in-memory value stored in ArchiveSize with it. Then, it will seek the
30// stream to the earlier captured position of the size value (where zeros were written initially), and it will write the size
31// value into the stream at that position. With all that done, it will seek back to the position of the stream before the
32// proxy was encountered, and resume serialization.
33template<typename TSize, typename TOffset>
34struct ArchiveSize {
35 using ValueType = TSize;
36
37 struct Proxy {
38 ArchiveSize* target;
39 Anchor<TOffset>* base;
40
41 Proxy(ArchiveSize& size, Anchor<TOffset>& offset) : target{std::addressof(size)}, base{std::addressof(offset)} {
42 target->proxy = this;
43 base->onMove(onBaseMoved, this);
44 }
45
46 ~Proxy() {
47 if (target != nullptr) {
48 target->proxy = nullptr;
49 }
50 if (base != nullptr) {
51 base->onMove(onBaseMoved, nullptr);
52 }
53 }
54
55 Proxy(const Proxy&) = delete;
56 Proxy& operator=(const Proxy&) = delete;

58 Proxy(Proxy&& rhs) : target{nullptr}, base{nullptr} {
59 std::swap(target, rhs.target);
60 std::swap(base, rhs.base);
61 target->proxy = this;
62 base->onMove(onBaseMoved, this);
63 }
64

66 std::swap(target, rhs.target);

68 target->proxy = this;

75 if (data != nullptr) {

77 proxy->base = anchor;
