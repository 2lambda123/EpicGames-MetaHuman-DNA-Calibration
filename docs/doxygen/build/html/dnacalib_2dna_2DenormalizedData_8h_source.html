30 void populateJointVariableAttributeIndices(const dna::Reader* source, LODMapping& destination) {
31 // Prepare storage for all available LODs
32 const auto lodCount = source->getLODCount();
33 destination.setLODCount(lodCount);
34 // Concatenate all output indices for each LOD
35 for (std::uint16_t i = 0u; i < source->getJointGroupCount(); ++i) {
36 const auto outputIndices = source->getJointGroupOutputIndices(i);
37 const auto lodSizes = source->getJointGroupLODs(i);
38 assert(lodSizes.size() == lodCount);
39 for (std::uint16_t lod = 0u; lod < lodCount; ++lod) {
40 // In this case, each LOD has a distinct set of indices, so the LOD and Index parameters
41 // are the same for all LODs
42 destination.addIndices(lod, outputIndices.data(), lodSizes[lod]);
43 destination.associateLODWithIndices(lod, lod);
44 }
45 }
46 }
47
48 void populateMeshBlendShapeMappingIndices(const dna::Reader* source, LODMapping& destination) {
49 // Prepare storage for all available LODs
50 const auto lodCount = source->getLODCount();
51 destination.setLODCount(lodCount);
52 // Include only those mapping indices which are present in the already filtered
53 // mesh and blendshape LOD mapping
54 for (std::uint16_t lod = 0u; lod < lodCount; ++lod) {
55 const auto meshIndices = source->getMeshIndicesForLOD(lod);
56 const auto blendShapeIndices = source->getBlendShapeChannelIndicesForLOD(lod);
57
58 auto isMappingNeeded = [&meshIndices, &blendShapeIndices](const dna::MeshBlendShapeChannelMapping& mapping) {
59 const bool meshNeeded = extd::contains(meshIndices, mapping.meshIndex);
60 const bool blendShapeNeeded = extd::contains(blendShapeIndices, mapping.blendShapeChannelIndex);
61 return (meshNeeded && blendShapeNeeded);
62 };
63
64 for (std::uint16_t i = 0u; i < source->getMeshBlendShapeChannelMappingCount(); ++i) {
65 const auto mapping = source->getMeshBlendShapeChannelMapping(i);
66 if (isMappingNeeded(mapping)) {
67 // In this case, each LOD has a distinct set of indices, so the LOD and Index parameters
68 // are the same for all LODs
69 destination.addIndices(lod, &i, static_cast<std::uint16_t>(1));
70 }
71 }
72
73 destination.associateLODWithIndices(lod, lod);
74 }
75 }
76
77};
